Подвиг 4. Допишите следующую программу. Необходимо все 8 байт (тип double) переменной value вывести в консоль
 в виде десятичных чисел, записанных в одну строчку через пробел, начиная с первого байта.

#include <stdio.h>
int main(void)
{
    double value;
    int i;
    scanf("%lf", &value);

    char *ptr = (char *)&value; // Приведение указателя к указателю типа char для того, чтобы можно было вывести байты

    for (i = 0; i < sizeof(double); i++) 
        printf("%d ", (int)ptr[i]); // Вывод каждого байта

    return 0;
}

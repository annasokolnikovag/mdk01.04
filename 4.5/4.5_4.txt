Подвиг 8. Продолжите программу. Объявите указатель ptr_a на переменную a и указатель ptr_b на переменную b. Выполните обмен значениями переменных a и b через указатели ptr_a и ptr_b.

 #include <stdio.h>

int main(void)
{
    short a, b, init_a, init_b;
    short *ptr_a, *ptr_b; // Объявление указателей на переменные a, b

    scanf("%hd, %hd", &a, &b);
    init_a = a; init_b = b;

    ptr_a = &a; // Присваивание указателям адреса переменных a и b
    ptr_b = &b;    

    short swap = *ptr_a; // Сохранение значения a во временной переменной 
    *ptr_a = *ptr_b;
    *ptr_b = swap;       // Обмен значений переменных через указатели и временную переменную
    
    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}

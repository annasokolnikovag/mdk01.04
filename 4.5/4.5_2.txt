Подвиг 6. Продолжите программу. Объявите указатель с именем ptr_k на переменную k. Увеличьте значение переменной k на 10, используя указатель ptr_k.

#include <stdio.h>

int main(void)
{
    int init, k;
    int *ptr_k; // Объявление указателя на k
    scanf("%d", &init);
    k = init;

    ptr_k = &k; // Присваивание указателю адреса k
    *ptr_k += 10; // Увеличение значения через указатель

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
